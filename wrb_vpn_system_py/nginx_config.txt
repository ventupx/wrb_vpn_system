server {
    listen 80;
    server_name best.kuajingweb.cc;  # 第一个域名

    root /var/www/client_1/dist;  # 第一个域名对应的静态文件目录
    index index.html index.htm;

    location / {
        try_files $uri $uri/ /index.html;  # 用于支持单页应用的路由
    }

    # 处理媒体文件请求
    location /media/ {
        alias /path/to/your/django/media/folder/;  # 替换为实际的media文件夹路径
        add_header Access-Control-Allow-Origin *;
    }

    # 代理后端请求
    location /api/ {
        proxy_pass http://127.0.0.1:8008;  # 后端服务的地址和端口
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
